/**************************常用标签*******************/

*****基本标签：

<html>   // 根标签

<head>      //头部标签


<title> 网页标题 </title>  // 标题标签，用于显示网页标题 


</head>  

<body>    </body>   //  主体标签



</html>  



******文本标签：

<h1> ...... <h6>  // 文本标题标签，共六个

<p>  </p>  // 段落标签，用于区分段落

<br />     // 换行标签，单标签



******图片标签(img):


<img src=" " alt=" " title=" " ....../>   // 图片标签用于显示图片及设置属性

 src    // 图片路径
 alt    // 替换文本，图片不能显示是用此文本代替
 title   // 提示文本，当鼠标悬浮在图片上时显示该文本、 
 
 width
 height      //  设置图像的宽高 
 border     // 设置图像的边框粗细
 

*******超链接标签：

<a href="链接地址"  target="新窗口打开方式">       文字             </a>  // 超链接标签

     href: “#”空链接 
             “http://www.baidu.com” 外部链接
               ".html"    内部链接
               ".zip"      下载链接   

    target: _slef 使用当前窗口打开新页面  _blank 使用新窗口打开新页面




*******注释标签：

    <!--   注释标签       -->      //   注释标签 




*******特殊字符：

    &nbsp；      //  空格

    &lt;    //  “<” 字符
    &gt   //   “>” 字符
    



********表格标签：

语法-->

   <table>        // 用于定义表格的标签，最外围标签


    <tr>   // 行标签，嵌套在<table>标签内
 
       <th>           </th>  // 表头单元格，内容居中加粗
      
      <td>    </td>     // 普通单元格标签                              
               
                
    </tr>   

    </table>   
    

<table>属性：

  align           // 表格位置(居中)

  width          
  height   // 表格宽高

  border   // 表格变宽线有无

<td>单元格属性：
  
  colspan="个数"     // 跨列合并单元格(个数)
  rowspan="个数"    // 跨行合并单元格



*************列表标签：

<ul>  // 无序列表

<li>          </li>  // 列项
<li>          </li>



</ul>


<ol>   // 有序列表


<li>          </li>  // 列项
<li>          </li>



</ol>



<dl>   // 自定义列表


<dt>             </dt>   // 表头

<dd>            </dd>  // 对表头的描述


</dl>




*****************表单标签：


<form action="提交地址" name="表单名" >

<input type="控件类型">     // 输入型表单元素




</form>


常用表单控件属性-->

<input type="text">            // 普通单行输入框
            type="password"    // 密码输入框
            type="radio"            // 单选控件
            type="checkbox"     // 复选框
            type="submit"          // 提交按钮(注册....)
            type="reset"            //  重置按钮        
            type="button"          // 普通按钮(获验证码.....)
            type="file"               // 上传文件按钮 

            name="名字"            // 控件名字
            value="值"
            check="checked"                 // 默认选择属性
            maxlength="6"              // 允许最大输入字符个数


<select>            // 下拉表单

<option> 元素1  </option>  //元素标签


<option> 元素2  </option>



</select>


<textarea>      //   多行文本表单


</textarea>

 

*****************label标签：

<label for="表单元素id">  input表单元素名称  </label>   // label标签通常和input表单元素一起使用
